# herbie-mlir library

include(${PROJECT_SOURCE_DIR}/cmake/TableGenHelpers.cmake)

# ============================================================================
# herbieMLIR Library

  HerbieMLIRPasses.cpp

  ${HERBIE_MLIR_SOURCES}
